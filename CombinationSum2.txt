class Solution {
    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> result = new ArrayList<>();
        if (null == candidates) return result;
        Arrays.sort(candidates);
        List<Integer> tmp = new ArrayList<>();
        getSum (candidates, target, 0, tmp, result);
        return result;
        
    }
    private void getSum (int[] candidates, int target, int start, List<Integer> tmp, List<List<Integer>> result){
        for (int i = start; i < candidates.length; i++){
            if (i > start && candidates[i] == candidates[i-1]) continue;// remove duplicate
            if (candidates[i] == target){
                tmp.add(candidates[i]);
                result.add(new ArrayList<Integer>(tmp));
                tmp.remove(tmp.size() -1);
            }else if (candidates[i] < target){
                tmp.add(candidates[i]);
                getSum(candidates,target -candidates[i],i+1,tmp,result);
                tmp.remove(tmp.size() -1);
            }else return;
        }
    }
}
